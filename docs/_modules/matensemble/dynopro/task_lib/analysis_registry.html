    @classmethod
    def register(cls, name: str):
        def decorator(func: Callable):
            @wraps(func)
            def wrapper(*args, **kwargs):
                return func(*args, **kwargs)
            cls._registry[name] = wrapper
            return wrapper
        return decorator
